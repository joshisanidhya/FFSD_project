UI HELPERS (TOAST NOTIFICATIONS) // ------------------------------------------------------------------ /** * Creates a temporary popup notification on the screen. * Replaces standard browser alert() boxes for a modern look. * * @param {string} message - The message to display. * @param {string} type - 'success', 'error', or 'info'. */ export function showToast(message, type = 'info') { // Check if a toast container exists, if not, create one let container = document.getElementById('toast-container'); if (!container) { container = document.createElement('div'); container.id = 'toast-container'; // Basic inline styles so it works without requiring extra CSS files Object.assign(container.style, { position: 'fixed', bottom: '20px', right: '20px', zIndex: '9999', display: 'flex', flexDirection: 'column', gap: '10px' }); document.body.appendChild(container); } // Create the toast element const toast = document.createElement('div'); toast.className = `toast toast-${type}`; toast.textContent = message; // Style the toast based on the type let bgColor = '#333'; // default info if (type === 'success') bgColor = '#4CAF50'; if (type === 'error') bgColor = '#F44336'; Object.assign(toast.style, { backgroundColor: bgColor, color: 'white', padding: '12px 20px', borderRadius: '5px', boxShadow: '0 4px 6px rgba(0,0,0,0.1)', fontFamily: 'sans-serif', fontSize: '14px', opacity: '0', transition: 'opacity 0.3s ease-in-out', minWidth: '250px' }); container.appendChild(toast); // Fade in requestAnimationFrame(() => { toast.style.opacity = '1'; }); // Remove after 3 seconds setTimeout(() => { toast.style.opacity = '0'; setTimeout(() => toast.remove(), 300); // Wait for fade out animation }, 3000); }
